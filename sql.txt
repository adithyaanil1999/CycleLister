CREATE TABLE s_login(
    reg_s varchar(15) PRIMARY KEY,
    password varchar(100));
CREATE TABLE sponsor(
    id_sponsor varchar(15) NOT null,
    name varchar(30),
    block varchar(2),
    email varchar(40),
    phone varchar(15),
	FOREIGN KEY (id_sponsor) REFERENCES s_login(reg_s)
    	ON DELETE CASCADE);
CREATE TABLE cycle_list(
    Chasis_no varchar(30),
    reg_no varchar(15),
    brand varchar(20),
    location varchar(5),
    available varchar(2),
    FOREIGN KEY (reg_no) REFERENCES s_login(reg_s)
    	ON DELETE CASCADE);
CREATE TABLE r_login(
    reg_r varchar(15) PRIMARY KEY,
    password varchar(100));
CREATE TABLE renter(
    id_renter varchar(15) NOT null,
    name varchar(30),
    block varchar(2),
    email varchar(40),
    phone varchar(15),
	FOREIGN KEY (id_renter) REFERENCES r_login(reg_r)
    	ON DELETE CASCADE);
ALTER TABLE `cycle_list` ADD UNIQUE(`Chasis_no`);
CREATE TABLE sponsor(
    id_sponsor varchar(15) NOT null PRIMARY KEY,
    name varchar(30),
    block varchar(2),
    email varchar(40),
    phone varchar(15),
    Wallet_sponsor int(10) NOT null DEFAULT 0,
	FOREIGN KEY (id_sponsor) REFERENCES s_login(reg_s)
    	ON DELETE CASCADE);
CREATE TABLE trans_cycle(
    id_trans int not null AUTO_INCREMENT PRIMARY KEY,
    t_diff int(40),
    total int(30),
    id_rec varchar(30) not null);
CREATE table wallet( id_wallet varchar(30) NOT null PRIMARY KEY, 
    money int(20) DEFAULT 0, withdraw_status varchar(2), 
    FOREIGN KEY (id_wallet) REFERENCES sponsor(id_sponsor));

ALTER TABLE `sponsor` ADD `wallet_sponsor` INT NOT NULL AFTER `phone`;
ALTER TABLE `trans_cycle` ADD `t_status` INT NOT NULL AFTER `id_rec`;
